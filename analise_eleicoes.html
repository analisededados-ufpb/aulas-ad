eleicoes_dif <- eleicoes %>%
  filter(nr_turno == 1, nm_urna_candidato == "JAIR BOLSONARO" | nr_candidato == 13) %>%
  group_by(uf, ano_eleicao) %>%
  summarise(
    perc_votos_jb = perc_votos[nm_urna_candidato == "JAIR BOLSONARO"],
    perc_votos_pt = perc_votos[nr_candidato == 13],
    dif_jb_pt = perc_votos_jb - perc_votos_pt
  ) %>%
  mutate(
    direcao = ifelse(dif_jb_pt[ano_eleicao == 2022] >= dif_jb_pt[ano_eleicao == 2018],
                     "cresceu", "diminuiu")
  )

eleicoes_dif_w <- eleicoes_dif %>%
  pivot_wider(id_cols = uf, names_from = ano_eleicao,
              values_from = dif_jb_pt, names_prefix = 'dif_') %>%
  ungroup() %>%
  mutate(
    direcao = ifelse(dif_2022 > dif_2018, "aumentou", "diminuiu"),
    dif_2022_2018 = round(dif_2022 - dif_2018),
    uf = fct_reorder(uf, dif_2022_2018, .desc = T)
  )
p <- ggplot(eleicoes_dif_w, aes(y = uf)) +
    geom_col(aes(dif_2022, fill = dif_2022), col = 'darkgray', show.legend = F) +
    geom_vline(aes(xintercept=0), lty=2) +
    geom_point(aes(dif_2018)) +
    geom_segment(aes(x=dif_2018, xend=dif_2022, yend=uf),
                 arrow = arrow(length = unit(0.05, "inches"))) +
